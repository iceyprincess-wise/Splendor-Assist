    private fun executeRollingRecord() {

        try {

            val runtimeFile =
                File(
                    cacheDir,
                    "dvr_runtime_${UUID.randomUUID()}.log"
                )

            FileOutputStream(runtimeFile).use { stream ->

                val payload =
                    "DVR ACTIVE :: ${System.currentTimeMillis()}"

                stream.write(
                    payload.toByteArray()
                )
            }

            RuntimeLogger.log(
                "DVR runtime artifact created",
                "DVR"
            )

        } catch (e: Exception) {

            RuntimeLogger.log(
                "DVR runtime artifact failed",
                "DVR"
            )
        }
    }
